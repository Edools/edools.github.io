def asset_url(asset_name, url_only: false)
  if RequestStore.store[:editor_mode]
    "/services/assets/#{current_theme_id}/editor_asset/#{asset_name}#{timestamp}"
  else
    opts = ['//', Rails.application.secrets.cdn_url, cdn_path, asset_name]
    opts.delete('//') if url_only

    File.join(*opts) + timestamp
  end
end
