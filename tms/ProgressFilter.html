def course_progress_card(enrollment, course = {})
  return enrollment['progress_card'] if enrollment['progress_card'].present? && course.blank?

  enrollment_id = enrollment['id']
  course_id     = course.fetch('id', nil)

  progress = CourseProgressCard.find_by_enrollment_and_course(enrollment_id, course_id) if enrollment_id

  return progress if progress.present?
  return build_progress_total(enrollment_id, course_id)
end

def get_statistics_data(enrollment, course)
  completed = 0
  grade = 0

  enrollment['lessons_progresses'].each do |progress|
    if progress['completed'] == true && progress['grade'].present?
      completed = completed + 1
      grade = grade + progress['grade']
    end
  end
  
  if completed != 0
    grade = grade / completed
  end

  {
    'completed' => completed,
    'grade' => grade
  }
end

def quiz_lesson_grade(enrollment, lesson_id)
  return 0 unless enrollment['lessons_progresses']

  enrollment['lessons_progresses'].each do |progress|
    return progress['grade'] if progress['lesson_id'] == lesson_id && progress['completed'] == true
  end

end
