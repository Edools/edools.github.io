def available_time_message enrollment
  if enrollment['unlimited']
    t "enrollment.available_time_message.unlimited"
  else
    date = enrollment['available_until'].strftime('%m/%d/%Y')
    t "enrollment.available_time_message.available_until", "date" => date
  end
end

def total_max_attendance_message enrollment
  unless attendance_type_indeterminate? enrollment
    if attendance_type_time? enrollment
      time = format_time(enrollment['max_attendance_length'])
      t "enrollment.total_max_attendance_message.attendance_type_time", "time" => time
    else
      attempts = enrollment['max_attendance_length']
      t "enrollment.total_max_attendance_message.attendance_type_attempts", "attempts" => attempts
    end
  end
end

def use_max_attendance_message enrollment
  unless attendance_type_indeterminate? enrollment
    if attendance_type_time? enrollment
      time = format_time(enrollment['time_spent'] || 0)
      t "enrollment.use_max_attendance_message.attendance_type_time", "time" => time
    else
      attempts = enrollment['lessons_views']
      t "enrollment.use_max_attendance_message.attendance_type_attempts", "attempts" => attempts
    end
  end
end
