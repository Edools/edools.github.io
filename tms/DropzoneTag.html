def initialize(tag_name, markup, options)
  super

  @dropzone_options = {}
  @type = markup.split(',')[0].strip

  markup.scan(Liquid::TagAttributes) do |key, value|
    @dropzone_options[key.to_sym] = value
  end
end

def render(context)
  @dropzone_type = context[@type]
  @school        = context['current_school']
  @resource      = resource_from(context)

  dropzone_variables

  context.stack do
    @html_block = render_all(@nodelist, context)
  end

  render_view('dropzone')
end
