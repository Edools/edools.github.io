- (Object) localize(input, opts = {})

Reformat a Time using Ruby's I18n

If the format was specified in the options, you need to have it in the theme's locale body. Like defaults locale, this format need to be scoped by time.formats.your-format-name.

Usage example: post.date | localize: "%d.%m.%Y" } post.date | localize: format: "short" }

def localize(input, opts = {})
  format = nil

  if opts['format']
    translations = RequestStore.store[:current_locale].parsed_body
    format       = translations['time']['formats'][opts['format']]
  elsif opts.present?
    format       = opts
  end

  I18n.l input, :format => format
end

def t(translation_key, args = {})
  locale = RequestStore.store[:current_locale]

  return 'Locale not found' unless locale

  translation = translation_key.split('.').reduce(locale.parsed_body) do |hash, key|
    hash && hash[key]
  end

  return "#{translation_key} is not present at current locale" unless translation

  if hash_with_values?(translation) && hash_with_key?(args, 'count')
    translation = handle_pluralization(translation_key, translation, args['count'])
  elsif hash?(translation)
    return "#{translation_key} is incomplete or you forgot to pass count argument"
  end

  if hash_with_values?(args)
    Liquid::Template.parse(translation).render(args)
  else
    translation
  end
end
