def add
  product = SchoolProduct.find_by(slug: params[:product_slug])

  if product
    payment_option = product.payment_options.find_by(id: params[:payment_option_id])
  end

  response = current_order.add(product, payment_option)

  flash[:alert] = "errors.cart.#{response}" unless response == :ok

  redirect_to cart_path
end

def apply_coupon
  token = current_order.apply_coupon(params[:coupon])

  if token && token.persisted?
    flash[:notice] = 'checkout.cart.coupon_applied'
  else
    flash[:alert] = 'errors.cart.can_not_apply_coupon'
  end

  redirect_to cart_path
end

def remove
