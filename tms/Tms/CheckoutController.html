def pay
  form = CheckoutPayForm.new(current_user, current_order,
                             current_school.active_credential, permitted_order_params)
  if form.submit
    remove_order_from_session!

    if form.external_redirection?
      if capybara_request?
        render text: "You will be redirected to #{form.payment.link}"
      else
        redirect_to form.payment.link
      end
    else
      redirect_to checkout_success_path(payment_id: form.payment)
    end
  else
    flash.now[:alert] = form.errors.details.stringify_keys
    # render payment
    payment
  end
end

def payment
  gateway_credential = current_school.active_credential
  limit = gateway_credential.installments_limit
  amount_limit = gateway_credential.installments_amount_limit

  render_template('checkout/payment',
    'order' => current_liquid_order,
    'gateway' => gateway_credential,
    'credit_card' => params[:order].try(:[], :credit_card),
    'active_payment_type' => params[:order].try(:[], :payment_type) || 'credit_card',
    'installments' => current_order.installments_options(limit, amount_limit)
  )
end
