def default_errors errors
  if errors
    html = ""

    if errors.is_a?(String)
      html << t(errors)
    elsif errors.is_a?(Hash) && errors.keys.include?(:translation_key) && errors.keys.include?(:translation_args)
      html << t(errors[:translation_key], errors[:translation_args])
    else
      errors.each { |k, v| html << "<li>#{format_errors(k, v)}</li>" }
    end

    html.html_safe
  end
end
