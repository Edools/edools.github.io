def content_path(content, options = {})
  enrollment_id = options.fetch('enrollment', 'id' => nil)['id']
  course_id = options.fetch('course', 'id' => nil)['id']

  path = enrollment_id ? "/enrollments/#{enrollment_id}" : "/admin_view"
  path << "/courses/#{course_id}/course_contents/#{content['id']}"
end

def course_path(enrollment, course_id=nil)
  course_id ||= enrollment['school_product']['library_resource']['resource_id']
  enrollment_id = enrollment['id']

  path = enrollment_id ? "/enrollments/#{enrollment_id}" : "/admin_view"
  path << "/courses/#{course_id}"
end
