def get_products_by_category_slug(school, category_slug, limit = 5)
  products = SchoolProduct
               .from_school(school['id'])
               .joins(:categories)
               .where(categories: { school_id: school['id'], slug: category_slug })
               .published
               .limit(limit)

  CollectionDrop.new products, { include: [:payment_options, { school_product_topics: { name: :topics } }] }
end
